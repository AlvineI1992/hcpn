FROM `referral_patientinfo`

AND `patientFirstName` = 'BOY'
AND `patientBirthDate` = '2014-08-23'
AND `patientSex` = 'M'
AND `refferalDate` = '2023-09-20'
AND `refferalTime` = '08:49:00'
AND `typeOfReferral` = 'TRANS'
AND `fhudFrom` = 'DOH000000000000499'
